-- ========== Binary Dumper ========== -- يقرأ eboot.bin و libc.prx ويرسلهما عبر الشبكة local target_ip = "192.168.8.1" -- ضع IP كمبيوترك هنا local port = 9027 local files = { {"/app0/eboot.bin", "eboot.bin"}, {"/app0/sce_module/libc.prx", "libc.prx"}, } local sock = syscall.socket(AF_INET, SOCK_STREAM, 0):tonumber() if sock < 0 then print("socket() error") return end local addr = memory.alloc(16) memory.write_byte(addr + 1, AF_INET) memory.write_word(addr + 2, bit32.bor(bit32.lshift(port, 8), bit32.rshift(port, 8))) -- ضع IP هنا (192.168.8.1 = 0xC0A80801) memory.write_dword(addr + 4, 0x0108A8C0) -- 192.168.8.1 معكوس local ret = syscall.connect(sock, addr, 16):tonumber() if ret < 0 then print("connect() error: "..ret) syscall.close(sock) return end print("Connected to "..target_ip..":"..port) for _, file_info in ipairs(files) do local src_path, dst_name = file_info[1], file_info[2] local fd = syscall.open(src_path, 0):tonumber() if fd < 0 then print("Cannot open "..src_path) else -- إرسال اسم الملف syscall.write(sock, dst_name.."\n", #dst_name + 1) local total = 0 local buf = memory.alloc(0x10000) -- 64KB buffer while true do local n = syscall.read(fd, buf, 0x10000):tonumber() if n <= 0 then break end local sent = 0 while sent < n do local s = syscall.write(sock, buf + sent, n - sent):tonumber() if s <= 0 then break end sent = sent + s end total = total + n if total % 0x100000 == 0 then print(dst_name..": "..total.." bytes...") end end syscall.close(fd) print(dst_name..": DONE ("..total.." bytes)") -- إرسال marker للملف التالي syscall.write(sock, "==END==\n", 8) end end syscall.close(sock) print("Dump complete!") notify("Binary dump complete!")
